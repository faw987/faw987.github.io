function shakeDetection(event) { const acceleration = event.accelerationIncludingGravity; if (lastX !== null && lastY !== null && lastZ !== null) { const deltaX = Math.abs(lastX - acceleration.x); const deltaY = Math.abs(lastY - acceleration.y); const deltaZ = Math.abs(lastZ - acceleration.z); if ((deltaX > shakeThreshold && deltaY > shakeThreshold) || (deltaX > shakeThreshold && deltaZ > shakeThreshold) || (deltaY > shakeThreshold && deltaZ > shakeThreshold)) { const currentTime = new Date(); if (currentTime - lastTime > 1000) { alert(`Shake detected!\n\nDeltaX: ${deltaX.toFixed(2)}\nDeltaY: ${deltaY.toFixed(2)}\nDeltaZ: ${deltaZ.toFixed(2)}`); lastTime = currentTime; } } } lastX = acceleration.x; lastY = acceleration.y; lastZ = acceleration.z; }
